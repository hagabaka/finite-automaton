  def deterministic?
    @transitions.keys.all? do |(from, character)|
      character != EPSILON || @transitions[character].empty?
    end and
    @states.all? do |state|
      @alphabet.all? do |character|
        @transitions[[state, character]].length == 1
      end
    end
  end
